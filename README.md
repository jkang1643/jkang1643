from openai import OpenAI

app = FastAPI()

@app.post("/ai-assist")
async def ai_assisted_development(prompt: str):
    client = OpenAI()
    response = client.chat.completions.create(
        model="gpt-4",
        messages=[{"role": "user", "content": prompt}]
    )
    return {"assistance": response.choices[0].message.content}
